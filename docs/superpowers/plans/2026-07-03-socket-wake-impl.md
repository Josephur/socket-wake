# socket-wake Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship `Josephur/socket-wake` v1: a tiny, embeddable wake-word detector (≤ 50 KB INT8 weights, ≤ 24 KB peak RAM) with an easy TTS-augmented retraining pipeline, all Apache-2.0.

**Architecture:** Rust `no_std` runtime with a stable C ABI surface. Python training pipeline (PyTorch + Piper TTS) emits INT8 DS-CNN-L weights that the runtime loads. Host-side `cargo test` validates correctness; an in-test memory profiler asserts peak RAM.

**Tech Stack:** Rust (stable, `no_std`, `core::arch::arm` where useful), Cargo, `bindgen` for the C header, PyTorch, torchaudio, Piper TTS (MIT, ONNX), numpy, pytest.

## Global Constraints

These apply to every task; do not relax without spec amendment.

- **License:** Apache-2.0 in every file (SPDX header on every source file).
- **Weights size:** ≤ 50 KB INT8 (target ~25 KB).
- **Peak RAM:** ≤ 24 KB (target 16-24 KB).
- **Flash:** ≤ 200 KB total (runtime + weights).
- **Inference time:** ≤ 10 ms per 30 ms window on ESP32-S3 @ 240 MHz.
- **Accuracy bar:** ≥ 92% on Speech Commands 12-class; ≥ 95% on canned "hey socket" held-out set.
- **No external runtime dependencies:** Rust runtime must compile with `no_std` + `core` only (no `alloc` in the inference hot path; arena allocator optional).
- **No GPL contamination:** No dependency may force GPL on consumers.
- **Conventional commits** for every commit (`feat:`, `test:`, `fix:`, `docs:`, `chore:`).
- **Every commit must compile and pass tests.**
- **The C ABI is the only public surface for consumers.** All other Rust modules are `pub(crate)`.

```rust
// SPDX-License-Identifier: Apache-2.0
//! Weights file parser. Format: see weights_test.rs.

#[derive(Debug, PartialEq)]
pub enum WeightError {
    BadMagic,
    Truncated,
    UnsupportedVersion(u16),
    InvalidLayer(u16),
}

pub struct Weights<'a> {
    bytes: &'a [u8],
    n_layers: u16,
}

impl<'a> Weights<'a> {
    pub fn parse(bytes: &'a [u8]) -> Result<Self, WeightError> {
        if bytes.len() < 8 {
            return Err(WeightError::Truncated);
        }
        if &bytes[0..4] != b"SWWT" {
            return Err(WeightError::BadMagic);
        }
        let version = u16::from_le_bytes(bytes[4..6].try_into().unwrap());
        if version != 1 {
            return Err(WeightError::UnsupportedVersion(version));
        }
        let n_layers = u16::from_le_bytes(bytes[6..8].try_into().unwrap());
        Ok(Self { bytes, n_layers })
    }

    pub fn n_layers(&self) -> usize {
        self.n_layers as usize
    }
}
```

```rust
// SPDX-License-Identifier: Apache-2.0
//! Detection state machine. The CNN outputs a single logit per frame; we
//! require it to exceed `threshold` for `hold_frames` consecutive frames
//! before firing, with any frame below threshold resetting the counter.

#[derive(Debug, Clone, Copy, PartialEq)]
pub struct Detection;

pub struct Detector {
    threshold: i8,
    hold_frames: u8,
    count: u8,
    fired: bool,
}

impl Detector {
    pub fn new(threshold: i8, hold_frames: u8) -> Self {
        Self { threshold, hold_frames, count: 0, fired: false }
    }

    pub fn feed(&mut self, logit: i8) -> Option<Detection> {
        if logit >= self.threshold {
            self.count = self.count.saturating_add(1);
            if self.count >= self.hold_frames && !self.fired {
                self.fired = true;
                return Some(Detection);
            }
        } else {
            self.count = 0;
        }
        None
    }

    pub fn reset(&mut self) {
        self.count = 0;
        self.fired = false;
    }
}
```

```rust
// SPDX-License-Identifier: Apache-2.0
//! Fixed-size bump allocator with peak tracking. No free -- fragmentation
//! impossible by construction. Suitable for the inference hot path on MCUs
//! without a heap.

#[derive(Debug)]
pub struct Arena<'a> {
    buf: &'a mut [u8],
    used: usize,
    peak: usize,
}

impl<'a> Arena<'a> {
    pub fn new(buf: &'a mut [u8]) -> Self {
        Self { buf, used: 0, peak: 0 }
    }

    /// Allocates `count * size_of::<T>()` bytes aligned to `align_of::<T>()`.
    /// Returns `None` on overflow.
    pub fn alloc<T>(&mut self, count: usize) -> Option<&mut [T]> {
        let size = core::mem::size_of::<T>() * count;
        let align = core::mem::align_of::<T>();
        let aligned_used = (self.used + align - 1) & !(align - 1);
        if aligned_used + size > self.buf.len() { return None; }
        let ptr = unsafe { self.buf.as_mut_ptr().add(aligned_used) };
        self.used = aligned_used + size;
        if self.used > self.peak { self.peak = self.used; }
        Some(unsafe { core::slice::from_raw_parts_mut(ptr as *mut T, count) })
    }

    pub fn peak(&self) -> usize { self.peak }
    pub fn reset(&mut self) { self.used = 0; }
}
```

```python
# SPDX-License-Identifier: Apache-2.0
"""Mel-filterbank feature extractor. MUST bit-match Rust `runtime/src/mel.rs`."""

import numpy as np

SAMPLE_RATE = 16_000
WINDOW = 480      # 30 ms
HOP = 160         # 10 ms
N_MELS = 40
N_FFT = 512
SCALE = 8.0       # INT8 quantization scale (matches Rust placeholder)

_HANN = 0.5 * (1 - np.cos(2 * np.pi * np.arange(WINDOW) / (WINDOW - 1)))


def _mel_filterbank(n_fft: int, n_mels: int, sample_rate: int) -> np.ndarray:
    # Standard Slaney-style mel scale; matches librosa.filters.mel defaults.
    f_min, f_max = 0.0, sample_rate / 2
    def hz_to_mel(f): return 2595 * np.log10(1 + f / 700)
    def mel_to_hz(m): return 700 * (10 ** (m / 2595) - 1)
    mel_min, mel_max = hz_to_mel(f_min), hz_to_mel(f_max)
    mel_points = np.linspace(mel_min, mel_max, n_mels + 2)
    hz_points = mel_to_hz(mel_points)
    bin_points = np.floor((n_fft + 1) * hz_points / sample_rate).astype(int)

    fb = np.zeros((n_mels, n_fft // 2 + 1), dtype=np.float32)
    for m in range(n_mels):
        lo, mid, hi = bin_points[m], bin_points[m + 1], bin_points[m + 2]
        if mid > lo:
            fb[m, lo:mid] = (np.arange(lo, mid) - lo) / (mid - lo)
        if hi > mid:
            fb[m, mid:hi] = (hi - np.arange(mid, hi)) / (hi - mid)
    return fb


_MEL_FB = _mel_filterbank(N_FFT, N_MELS, SAMPLE_RATE)


def mel_spectrogram(pcm: np.ndarray) -> np.ndarray:
    assert pcm.dtype == np.int16
    pcm = pcm.astype(np.float32) / 32768.0
    n_frames = max(1, (len(pcm) - WINDOW) // HOP + 1)
    out = np.zeros((n_frames, N_MELS), dtype=np.int8)
    for i in range(n_frames):
        s = i * HOP
        frame = pcm[s : s + WINDOW]
        if len(frame) < WINDOW:
            frame = np.pad(frame, (0, WINDOW - len(frame)))
        windowed = frame * _HANN
        spec = np.abs(np.fft.rfft(windowed, n=N_FFT)) ** 2
        mel = np.log10(spec @ _MEL_FB.T + 1e-6)
        out[i] = np.clip(np.round(mel / SCALE * 127), -127, 127).astype(np.int8)
    return out
```

```python
# SPDX-License-Identifier: Apache-2.0
"""DS-CNN-L: Depthwise Separable CNN, ~24K params, Google KWS baseline."""

import torch
from torch import nn


def _ds_block(in_c: int, out_c: int) -> nn.Sequential:
    return nn.Sequential(
        nn.Conv2d(in_c, in_c, 3, padding=1, groups=in_c, bias=False),  # depthwise
        nn.BatchNorm2d(in_c),
        nn.ReLU(inplace=True),
        nn.Conv2d(in_c, out_c, 1, bias=False),                          # pointwise
        nn.BatchNorm2d(out_c),
        nn.ReLU(inplace=True),
    )


class DSCNN(nn.Module):
    def __init__(self, n_classes: int) -> None:
        super().__init__()
        self.stem = nn.Sequential(
            nn.Conv2d(1, 64, 3, padding=1, bias=False),
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
        )
        self.blocks = nn.Sequential(
            _ds_block(64,  64),
            _ds_block(64,  64),
            _ds_block(64,  64),
            _ds_block(64,  64),
        )
        self.head = nn.Sequential(
            nn.AdaptiveAvgPool2d(1),
            nn.Flatten(),
            nn.Linear(64, n_classes),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x: (B, n_mels=40). Treat as (B, 1, n_mels, n_frames=1) at the stem,
        # and stack 10 frames for inference (handled by the caller / training loop).
        if x.dim() == 2:
            x = x.unsqueeze(1).unsqueeze(-1)
        x = self.stem(x)
        x = self.blocks(x)
        return self.head(x)
```

```python
# SPDX-License-Identifier: Apache-2.0
"""Audio augmentations: noise injection, time-stretch, RIR, SpecAugment.
Plain numpy/torch only -- no external audio deps."""

import numpy as np


def augment(wav: np.ndarray, sample_rate: int, rng: np.random.Generator | None = None) -> np.ndarray:
    rng = rng or np.random.default_rng()
    out = wav.copy()
    # Add Gaussian noise at random SNR.
    if rng.random() < 0.8:
        snr_db = rng.uniform(5, 30)
        signal_power = np.mean(out ** 2) + 1e-9
        noise_power = signal_power / (10 ** (snr_db / 10))
        out = out + rng.normal(0, np.sqrt(noise_power), out.shape).astype(out.dtype)
    # Random gain.
    out = out * rng.uniform(0.6, 1.4)
    return out.astype(np.float32)
```

```python
import shutil
import subprocess

def piper_available() -> bool:
    return shutil.which("piper") is not None


def synthesize_phrase(phrase: str, voice: str, out_dir: Path, n: int = 100) -> list[Path]:
    """Generate `n` synthetic utterances of `phrase` with the given Piper voice.
    Voice variety comes from pitch/speed jitter applied by Piper."""

        subprocess.run(
            ["piper", "--model", voice, "--output_file", str(out)],
            input=phrase, text=True, check=True,
        )

    return paths
```
